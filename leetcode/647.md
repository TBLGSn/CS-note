//以当前为中心向两边扩展
class Solution {
public:
    int extend(const string& s, int left, int right){
        int n = s.size();
        int res = 0;
        while (left>=0 && right < n && s[left] == s[right])
        {
            left--;
            right++;
            res  ++;
        }
        return res;
    }
     int countSubstrings(string s) {
       int res = 0;
       for(int i = 0,n = s.size(); i < n; ++i) {
           res += extend(s, i, i);
           res += extend(s, i,i+1);
       }
        return res;
    }
};
